Navigator API


                    // CSS 3 selectors
                    $("div.some-class p:first[title='something']")

                    // Find via index and/or attribute matching
                    $("h1", 2, class: "heading")
                    $("p", name: "description")
                    $("ul.things li", 2)

                    // 'text' is special attribute for the element text content
                    $("h1", text: "All about Geb")

                    // Chaining
                    $("div").find(".b")
                    $("div").filter(".c").parents()
                    $("p.c").siblings()

Page Object pattern


                    class LoginPage extends Page {
                        static url = "http://myapp.com/login"
                        static at = { heading.text() == "Please Login" }
                        static content = {
                        heading { $("h1") }
                        loginForm { $("form.login") }
                        loginButton(to: AdminPage) { loginForm.login() }
                        }
                    }

                    class AdminPage extends Page {
                        static at = { heading.text() == "Admin Section" }
                        static content = {
                        heading { $("h1") }
                        }
                    }

                    Browser.drive {
                        to LoginPage
                        assert at(LoginPage)
                        loginForm.with {
                        username = "admin"
                        password = "password"
                        }
                        loginButton.click()
                        assert at(AdminPage)
                    }

conclusion

  • pro:
    geb可模拟用户的鼠标操作，是有效的 web 项目自动化测试方案。
  • con:
    需要测试人员通过编程实现，并具备 groovy + jquery 基础。
